üíª Agent Creation Example

# Simple Agent Creation with All Three Components
from langchain_ollama import ChatOllama
from langchain.agents import create_agent
import tools

# 1. LLM: Define the language model
model = ChatOllama(model="qwen3", temperature=0.1)

# 2. TOOLS: Select tools for the agent
agent_tools = [tools.web_search, tools.calculate, tools.analyze_data]

# 3. PROMPT: Define behavior (optional, can be string or callable)
system_prompt = "You are a helpful research assistant."

# Create Agent: Combining LLM + Tools + Prompt
agent = create_agent(
    model,           # LLM component
    tools=agent_tools,   # Tools component  
    prompt=system_prompt # Prompt component
)

# Use the agent
result = agent.invoke({
    "messages": "Search for AI news and calculate 25 * 4"
})
